1function euclides(a, b) {
2  let tempA = Math.abs(a);
3  let tempB = Math.abs(b);
4  while (tempB > 0) {
5    let q = Math.floor(tempA / tempB);
6    let r = tempA % tempB;
7    tempA = tempB;
8    tempB = r;
9  }
10  return tempA;
11}
1function euclidesExtendido(A, B) {
2  let r = [Math.abs(A), Math.abs(B)];
3  let x = [1, 0], y = [0, 1];
4  let i = 1;
5  while (r[i] !== 0) {
6    let q = Math.floor(r[i-1] / r[i]);
7    r.push(r[i-1] % r[i]);
8    x.push(x[i-1] - q * x[i]);
9    y.push(y[i-1] - q * y[i]);
10    i++;
11  }
12  let xFinal = A < 0 ? -x[i-1] : x[i-1];
13  let yFinal = B < 0 ? -y[i-1] : y[i-1];
14  return { gcd: r[i-1], x: xFinal, y: yFinal };
15}

¿Sabías que...? (Aplicación en Criptografía)

El Algoritmo de Euclides Extendido es una pieza fundamental en la criptografía moderna de clave pública, como el protocolo RSA.

Para cifrar o firmar mensajes, RSA necesita calcular un Inverso Multiplicativo Modular. Es decir, dado un exponente de cifrado e y el valor de la función φ(n), se busca una clave privada d tal que:

e · d ≡ 1 (mod φ(n))

Esto significa que e · d - k · φ(n) = 1. ¡Esta es exactamente la Identidad de Bézout! Si resolvemos el algoritmo extendido para e y φ(n), obtenemos el coeficiente d como el inverso de e.
